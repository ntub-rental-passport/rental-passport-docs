' ════════════════════════════════════════════════════════════
'  Frontend Package
' ════════════════════════════════════════════════════════════
package "Frontend Package" as Frontend {
    package "router/"     as router
    package "views/"      as views
    package "components/" as components
    package "stores/"     as stores
    package "services/"   as fe_svc

    ' 強制水平排列
    router     -[hidden]right- views
    views      -[hidden]right- components
    components -[hidden]right- stores
    stores     -[hidden]right- fe_svc
}

' ════════════════════════════════════════════════════════════
'  Backend Package（重整：3 列垂直結構，ai/ 為 services/ 同層）
' ════════════════════════════════════════════════════════════
package "Backend Package" as Backend {

    ' ── 第 1 列：api ─────────────────────────────
    package "api/"      as api

    ' ── 第 2 列：services + ai（並排）────────────
    together {
        package "services/" as be_svc
        package "ai/"       as ai_module
    }

    ' ── 第 3 列：models + schemas + core（並排）──
    together {
        package "models/"  as models
        package "schemas/" as schemas
        package "core/"    as core
    }

    ' ── 強制三層垂直堆疊 ─────────────────────────
    api    -[hidden]down- be_svc
    be_svc -[hidden]down- models

    ' ── 強制第 2、3 列水平排列 ───────────────────
    be_svc  -[hidden]right- ai_module
    models  -[hidden]right- schemas
    schemas -[hidden]right- core
}

' ════════════════════════════════════════════════════════════
'  Database Package
' ════════════════════════════════════════════════════════════
package "Database Package" as Database {
    package "MySQL\n(關聯式資料庫)"  as mysql
    package "ChromaDB\n(向量資料庫)" as chroma
    mysql -[hidden]down- chroma
}

' ════════════════════════════════════════════════════════════
'  External Package
' ════════════════════════════════════════════════════════════
package "External Package" as External {
    package "Google Cloud Vision"  as vision
    package "Anthropic Claude API" as claude
    package "Google OAuth API"     as oauth
    package "Web Push Service"     as webpush
    package "政府開放資料"          as govdata

    vision  -[hidden]down- claude
    claude  -[hidden]down- oauth
    oauth   -[hidden]down- webpush
    webpush -[hidden]down- govdata
}

' ── Database 與 External 同列並排 ────────────
Database -[hidden]right- External

' ── 整體垂直層次：Frontend ▼ Backend ▼ (Database + External) ──
Frontend -[hidden]down- Backend
Backend  -[hidden]down- Database

' ════════════════════════════════════════════════════════════
'  關係（極簡：Backend 內部 3 條，全部垂直向下）
' ════════════════════════════════════════════════════════════

' ── Frontend 內部 ──
router  ..> views   : «import»
views   ..> stores  : «access»
stores  ..> fe_svc  : «access»

' ── 前端 → 後端 ──
fe_svc  ..> api : HTTP / REST

' ── Backend 內部（只留 3 條主要呼叫鏈，全部垂直向下） ──
api     ..> be_svc    : «import»
be_svc  ..> ai_module : «access»
be_svc  ..> models    : «access»

' ── Backend → Database ──
models    ..> mysql  : «access»
ai_module ..> chroma : «access»

' ── External 整合（AI 相關保留 2 條、其他合併成 1 條） ──
ai_module ..> vision   : «access»
ai_module ..> claude   : «access»
be_svc    ..> External : «access»

' ── Web Push 反向推播 ──
webpush ..> Frontend : «push»
